1.1

1.3

كيف تُشغّل JavaScript؟

ثلاث طرق لتشغيل JavaScript في صفحة HTML:

HTML 
<!-- الطريقة 1: داخل وسم <script> في HTML -->
<!DOCTYPE html>
<html lang="ar">
<head>
  <title>أول برنامج JS</title>
</head>
<body>
  <h1 id="title">مرحباً</h1>

  <!-- الطريقة 1: في نهاية body مباشرة -->
  <script>
    console.log('مرحباً بالعالم!');
    alert('هذا هو JavaScript!');
  </script>
</body>
</html>

<!-- الطريقة 2: ملف خارجي (المُفضَّل) -->
<script src="js/main.js"></script>

<!-- الطريقة 3: Inline (غير مُنصح به) -->
<button onclick="alert('مرحبا');">اضغط هنا</button>
💡 لماذا في نهاية <body>؟

عند وضع <script> في نهاية <body>، يتحمّل HTML أولاً ثم يُنفَّذ JavaScript.

هذا يضمن أن جميع العناصر موجودة عند تنفيذ الكود.

البديل الحديث: إضافة defer أو async على الوسم: <script src='main.js' defer>

الجزء الثاني: المتغيرات (Variables)

المتغير هو صندوق له اسم نخزّن فيه قيمة. مثل: اسم المستخدم، عمره، رصيده، سلة مشترياته.

2.1

let وconst وvar

JavaScript 
// ══ const — للقيم الثابتة التي لا تتغير ══
const PI = 3.14159;
const SITE_NAME = 'أكاديمية التقنية';
const MAX_USERS = 100;
// PI = 3.14; // ❌ خطأ! لا يمكن تغيير const

// ══ let — للقيم التي قد تتغير ══
let userName = 'أحمد';
let score = 0;
let isLoggedIn = false;

userName = 'محمد';  // ✅ يمكن تغييرها
score = score + 10; // ✅ تحديث القيمة

// ══ var — القديم، تجنّبه ══
var oldVar = 'تجنّب استخدام var في الكود الحديث';
// var له مشاكل في النطاق (Scope) سنشرحها لاحقاً
الكلمةيمكن تغييرها؟متى تستخدمها؟
const❌ لاالقيم الثابتة (الأكثر استخداماً) ✅
let✅ نعمالقيم التي ستتغير
var✅ نعم❌ تجنّبه — مشاكل في النطاق

الجزء الرابع: العمليات والعوامل (Operators)

4.1

العمليات الحسابية (Arithmetic)

JavaScript 
const a = 10;
const b = 3;

console.log(a + b);   // 13  — الجمع
console.log(a - b);   // 7   — الطرح
console.log(a * b);   // 30  — الضرب
console.log(a / b);   // 3.333... — القسمة
console.log(a % b);   // 1   — باقي القسمة (Modulo)
console.log(a ** b);  // 1000 — الأس (10³)
console.log(Math.floor(a / b)); // 3 — قسمة صحيحة

// عمليات الزيادة والنقصان
let count = 0;
count++;         // count = 1 (Post-increment)
++count;         // count = 2 (Pre-increment)
count--;         // count = 1 (Post-decrement)
--count;         // count = 0 (Pre-decrement)

// الفرق بين Pre وPost
let x = 5;
console.log(x++); // 5 — يُطبع أولاً ثم يزيد
console.log(x);   // 6
console.log(++x); // 7 — يزيد أولاً ثم يُطبع
4.2

عوامل الإسناد (Assignment)

JavaScript 
let score = 100;

score = 100;    // إسناد عادي
score += 10;    // score = score + 10  → 110
score -= 5;     // score = score - 5   → 105
score *= 2;     // score = score * 2   → 210
score /= 3;     // score = score / 3   → 70
score %= 4;     // score = score % 4   → 2
score **= 3;    // score = score ** 3  → 8

// Nullish Coalescing Assignment (ES2021)
let name = null;
name ??= 'مجهول';  // يُسند فقط إذا كانت null أو undefined
console.log(name); // 'مجهول'

// Logical OR Assignment
let user = '';
user ||= 'زائر';   // يُسند إذا كانت falsy
console.log(user); // 'زائر'

4.4

العوامل المنطقية (Logical)

JavaScript 
const isLoggedIn = true;
const isAdmin    = false;
const age        = 20;

// && (AND) — كلاهما يجب أن يكون true
console.log(true && true);   // true
console.log(true && false);  // false
console.log(isLoggedIn && isAdmin); // false

// || (OR) — أحدهما كافٍ ليكون true
console.log(false || true);  // true
console.log(false || false); // false
console.log(isLoggedIn || isAdmin); // true

// ! (NOT) — عكس القيمة
console.log(!true);          // false
console.log(!isLoggedIn);    // false
console.log(!isAdmin);       // true

// ?? (Nullish Coalescing) — بديل null/undefined فقط
const userName = null ?? 'زائر';    // 'زائر'
const score    = 0    ?? 'لا يوجد'; // 0 (ليس null!)

// Short-circuit Evaluation — تقييم قصير
// && يُعيد الأول إذا كان falsy، وإلا يُعيد الثاني
console.log(null && 'نص');    // null
console.log('نص' && 42);     // 42

// || يُعيد الأول إذا كان truthy، وإلا يُعيد الثاني
const name = '' || 'افتراضي'; // 'افتراضي'
const port = process?.env?.PORT || 3000; // 3000

نطاق الدالة والإغلاق (Closure)

JavaScript 
// Closure: دالة داخلية تتذكر متغيرات الدالة الخارجية
function makeCounter(start = 0) {
  let count = start; // هذا المتغير محفوظ في الـ Closure

  return {
    increment: () => ++count,
    decrement: () => --count,
    reset:     () => { count = start; return count; },
    getValue:  () => count,
  };
}

const counter = makeCounter(10);
console.log(counter.getValue());   // 10
console.log(counter.increment());  // 11
console.log(counter.increment());  // 12
console.log(counter.decrement());  // 11
console.log(counter.reset());      // 10

// كل counter مستقل!
const counter2 = makeCounter(0);
counter2.increment();
counter2.increment();
console.log(counter2.getValue());  // 2
console.log(counter.getValue());   // 10 (لم يتغير)

// ══ الدالة 1: تحديد التقدير ══
function getGrade(score) {
  if (score >= 90) return { grade: 'A', label: 'ممتاز', emoji: '🏆' };
  if (score >= 80) return { grade: 'B', label: 'جيد جداً', emoji: '⭐' };
  if (score >= 70) return { grade: 'C', label: 'جيد', emoji: '👍' };
  if (score >= 60) return { grade: 'D', label: 'مقبول', emoji: '📚' };
  return { grade: 'F', label: 'راسب', emoji: '❌' };
}

// ══ الدالة 2: حساب المتوسط ══
const calcAverage = (arr) => {
  const total = arr.reduce((sum, s) => sum + s.score, 0);
  return (total / arr.length).toFixed(2);
};

// ══ الدالة 3: إيجاد الأعلى والأدنى ══
function getTopBottom(arr) {
  const sorted = [...arr].sort((a, b) => b.score - a.score);
  return {
    top:    sorted[0],
    bottom: sorted[sorted.length - 1],
  };
}

// ══ الدالة 4: تقرير الطالب ══
function printStudentReport(student) {
  const { grade, label, emoji } = getGrade(student.score);
  console.log('─'.repeat(40));
  console.log(`👤 الطالب: ${student.name}`);
  console.log(`📊 الدرجة: ${student.score}/100`);
  console.log(`🎯 التقدير: ${grade} — ${label} ${emoji}`);
}

// ══ الدالة 5: تقرير الفصل ══
function printClassReport(students) {
  console.log('═'.repeat(50));
  console.log('         📋 تقرير الفصل الدراسي');
  console.log('═'.repeat(50));

  // طباعة جميع الطلاب
  for (const student of students) {
    printStudentReport(student);
  }

  // إحصائيات عامة
  const avg = calcAverage(students);
  const { top, bottom } = getTopBottom(students);
  const passed  = students.filter(s => s.score >= 60).length;
  const failed   = students.length - passed;

  console.log('═'.repeat(50));
  console.log('         📈 الإحصائيات العامة');
  console.log('═'.repeat(50));
  console.log(`👥 عدد الطلاب:   ${students.length}`);
  console.log(`📊 متوسط الدرجات: ${avg}`);
  console.log(`✅ الناجحون:     ${passed} طالب`);
  console.log(`❌ الراسبون:     ${failed} طالب`);
  console.log(`🏆 الأعلى:       ${top.name} (${top.score})`);
  console.log(`📉 الأدنى:       ${bottom.name} (${bottom.score})`);
}

// ══ الدالة 6: إضافة طالب ══
function addStudent(students, name, score) {
  if (!name || name.trim() === '') {
    console.error('❌ الاسم مطلوب!');
    return students;
  }
  if (score < 0 || score > 100) {
    console.error('❌ الدرجة يجب أن تكون بين 0 و100!');
    return students;
  }
  const newStudent = {
    id:    students.length + 1,
    name:  name.trim(),
    score: score,
  };
  return [...students, newStudent];
}

// ══ تشغيل البرنامج ══
let classList = [...students];

// إضافة طلاب جدد
classList = addStudent(classList, 'محمد أحمد', 83);
classList = addStudent(classList, '', 90);    // ❌ خطأ
classList = addStudent(classList, 'هنا', 110); // ❌ خطأ

// طباعة التقرير الكامل
printClassReport(classList);

// تصفية الممتازين
const topStudents = classList.filter(s => s.score >= 90);
console.log('\n🏆 قائمة الشرف:');
topStudents.forEach(s => console.log(`  ★ ${s.name}: ${s.score}`));
