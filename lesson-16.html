ES2023: Array Immutable Methods

ES2023 أضافت طرقاً جديدة للمصفوفات لا تُعدّل الأصل — أكثر أماناً للبرمجة الوظيفية:

JavaScript 
const arr = [3, 1, 4, 1, 5, 9, 2, 6];

// ══ toSorted() — ترتيب بدون تعديل الأصل ✅ ══
const sorted = arr.toSorted((a, b) => a - b);
console.log(sorted); // [1, 1, 2, 3, 4, 5, 6, 9]
console.log(arr);    // [3,1,4,1,5,9,2,6] — لم يتغير!

// ══ toReversed() — عكس بدون تعديل ✅ ══
const reversed = arr.toReversed();
console.log(reversed); // [6,2,9,5,1,4,1,3]
console.log(arr);      // لم يتغير

// ══ toSpliced() — حذف/إضافة بدون تعديل ✅ ══
// toSpliced(start, deleteCount, ...items)
const spliced = arr.toSpliced(2, 2, 99, 88);
// يحذف 2 عناصر من الفهرس 2 ويُضيف 99 و88
console.log(spliced); // [3,1,99,88,5,9,2,6]
console.log(arr);     // لم يتغير

// ══ with() — تغيير عنصر بالفهرس بدون تعديل ✅ ══
const updated = arr.with(2, 100);
// يستبدل الفهرس 2 بقيمة 100
console.log(updated); // [3,1,100,1,5,9,2,6]
console.log(arr);     // لم يتغير

// with() مع فهرس سالب
const lastReplaced = arr.with(-1, 999);
// يستبدل آخر عنصر
console.log(lastReplaced); // [3,1,4,1,5,9,2,999]

// ══ findLast() وfindLastIndex() ══
const nums = [1, 2, 3, 4, 3, 2, 1];

// find من البداية
console.log(nums.find(n => n > 2));          // 3 (أول مطابق)
// findLast من النهاية
console.log(nums.findLast(n => n > 2));      // 3 (آخر مطابق)

console.log(nums.findIndex(n => n > 2));     // 2
console.log(nums.findLastIndex(n => n > 2)); // 4

// ══ Array.from() مع at() ══
const letters = ['a', 'b', 'c', 'd', 'e'];
console.log(letters.at(0));   // 'a'
console.log(letters.at(-1));  // 'e' ✅ (الأخير)
console.log(letters.at(-2));  // 'd'

ES2023: Symbols كـ WeakMap Keys

JavaScript 
// الآن يمكن استخدام Symbol كمفتاح في WeakMap وWeakSet

const key = Symbol('myKey');
const weakMap = new WeakMap();
weakMap.set(key, { secret: 'قيمة مخفية' });

// مفيد للبيانات الخاصة بدون حاجة لـ Object
const PRIVATE = Symbol('private');
const storage = new WeakMap();

function withPrivate(target, data) {
  storage.set(PRIVATE, { target, data });
  return target;
}

ES2024: Promise.withResolvers()

JavaScript 
// Promise.withResolvers() — إنشاء Promise مع تحكم خارجي
// بديل أنيق لـ: new Promise((resolve, reject) => { ... })

// ══ الطريقة القديمة ══
let resolveOld, rejectOld;
const promiseOld = new Promise((res, rej) => {
  resolveOld = res;
  rejectOld  = rej;
});

// ══ الطريقة الجديدة — أنيق ✅ ══
const { promise, resolve, reject } = Promise.withResolvers();

// استخدام عملي: ربط Promise بحدث
function waitForEvent(element, eventName) {
  const { promise, resolve, reject } = Promise.withResolvers();
  element.addEventListener(eventName, resolve, { once: true });
  return promise;
}

const btn = document.querySelector('#confirm-btn');
await waitForEvent(btn, 'click');
console.log('تم الضغط على الزر!');

// مثال: Dialog مع Promise
function showConfirmDialog(message) {
  const { promise, resolve } = Promise.withResolvers();
  const dialog = document.createElement('dialog');
  dialog.innerHTML = `
    <p>${message}</p>
    <button id='yes'>نعم</button>
    <button id='no'>لا</button>
  `;
  document.body.appendChild(dialog);
  dialog.showModal();
  dialog.querySelector('#yes').onclick = () => { resolve(true);  dialog.remove(); };
  dialog.querySelector('#no').onclick  = () => { resolve(false); dialog.remove(); };
  return promise;
}

const confirmed = await showConfirmDialog('هل تريد الحذف؟');
if (confirmed) deleteItem();

// ══ 2. Memoization — تخزين نتائج الحسابات الثقيلة ══
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

const slowFib = n => n <= 1 ? n : slowFib(n-1) + slowFib(n-2);
const fastFib = memoize(n => n <= 1 ? n : fastFib(n-1) + fastFib(n-2));
console.log(fastFib(50)); // سريع جداً!

// ══ 3. Debounce — تأخير تنفيذ الدالة ══
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

const searchInput = document.querySelector('#search');
const handleSearch = debounce(async (query) => {
  const results = await searchAPI(query);
  displayResults(results);
}, 300);

searchInput.addEventListener('input', e => handleSearch(e.target.value));

// ══ 4. Throttle — تقييد معدل التنفيذ ══
function throttle(fn, interval) {
  let lastTime = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      return fn.apply(this, args);
    }
  };
}

const handleScroll = throttle(() => {
  updateParallax();
}, 16); // ~60fps

window.addEventListener('scroll', handleScroll);

// ═══ js/models/Student.js ═══

export class Student {
  static #idCounter = 1000;
  static #instances = new Map();

  #id;
  #name;
  #email;
  #phone;
  #enrolledAt;
  #grades = new Map(); // courseId → Grade[]

  constructor({ name, email, phone }) {
    this.#id        = `STU-${++Student.#idCounter}`;
    this.name       = name;
    this.email      = email;
    this.phone      = phone;
    this.#enrolledAt = new Date();
    Student.#instances.set(this.#id, this);
  }

  get id()         { return this.#id; }
  get name()       { return this.#name; }
  get email()      { return this.#email; }
  get enrolledAt() { return this.#enrolledAt.toLocaleDateString('ar-SA'); }

  set name(val) {
    if (!val?.trim() || val.length < 2) throw new Error('الاسم قصير جداً');
    this.#name = val.trim();
  }
  set email(val) {
    if (!val?.includes('@')) throw new Error('البريد غير صالح');
    this.#email = val.toLowerCase().trim();
  }
  set phone(val) {
    if (!/^05[0-9]{8}$/.test(val)) throw new Error('الجوال غير صالح (يجب أن يبدأ بـ 05)');
    this.#phone = val;
  }

  addGrade(courseId, score, comment = '') {
    if (score < 0 || score > 100) throw new RangeError('الدرجة بين 0 و100');
    if (!this.#grades.has(courseId)) this.#grades.set(courseId, []);
    this.#grades.get(courseId).push({
      score, comment,
      date: new Date().toLocaleDateString('ar-SA'),
    });
  }

  getAverage(courseId) {
    const grades = this.#grades.get(courseId) ?? [];
    if (grades.length === 0) return null;
    return grades.reduce((s, g) => s + g.score, 0) / grades.length;
  }

  getOverallAverage() {
    const allGrades = [...this.#grades.values()].flat();
    if (allGrades.length === 0) return 0;
    return allGrades.reduce((s, g) => s + g.score, 0) / allGrades.length;
  }

  getLetterGrade() {
    const avg = this.getOverallAverage();
    if (avg >= 90) return { letter:'A', label:'ممتاز',  emoji:'🏆' };
    if (avg >= 80) return { letter:'B', label:'جيد جداً',emoji:'⭐' };
    if (avg >= 70) return { letter:'C', label:'جيد',    emoji:'👍' };
    if (avg >= 60) return { letter:'D', label:'مقبول',  emoji:'📚' };
    return           { letter:'F', label:'راسب',   emoji:'❌' };
  }

  toJSON() {
    return {
      id:          this.#id,
      name:        this.#name,
      email:       this.#email,
      phone:       this.#phone,
      enrolledAt:  this.#enrolledAt.toISOString(),
      grades:      Object.fromEntries(this.#grades),
    };
  }

  static fromJSON(data) {
    const student = new Student(data);
    // استعادة الـ id والدرجات
    student.#id = data.id;
    student.#enrolledAt = new Date(data.enrolledAt);
    Object.entries(data.grades ?? {}).forEach(([courseId, grades]) => {
      student.#grades.set(courseId, grades);
    });
    return student;
  }

  static getById(id)    { return Student.#instances.get(id); }
  static getAll()       { return [...Student.#instances.values()]; }
  static getTotalCount(){ return Student.#instances.size; }

  toString() {
    const { letter, emoji } = this.getLetterGrade();
    return `[${this.#id}] ${this.#name} | ${letter} ${emoji} | ${this.getOverallAverage().toFixed(1)}`;
  }
}

// ═══ js/services/StorageService.js ═══

export class StorageService {
  #prefix;

  constructor(prefix = 'techacademy_') {
    this.#prefix = prefix;
  }

  #key(name) { return `${this.#prefix}${name}`; }

  save(name, data) {
    try {
      localStorage.setItem(this.#key(name), JSON.stringify({
        data,
        savedAt: new Date().toISOString(),
        version: '1.0',
      }));
      return true;
    } catch (err) {
      console.error('فشل الحفظ:', err.message);
      return false;
    }
  }

  load(name, defaultValue = null) {
    try {
      const raw = localStorage.getItem(this.#key(name));
      return raw ? JSON.parse(raw).data : defaultValue;
    } catch {
      return defaultValue;
    }
  }

  remove(name)  { localStorage.removeItem(this.#key(name)); }
  exists(name)  { return localStorage.getItem(this.#key(name)) !== null; }
  getSavedAt(name) {
    const raw = localStorage.getItem(this.#key(name));
    return raw ? JSON.parse(raw).savedAt : null;
  }

  getAllKeys() {
    return Object.keys(localStorage)
      .filter(k => k.startsWith(this.#prefix))
      .map(k => k.slice(this.#prefix.length));
  }
}

// ═══ js/services/ApiService.js ═══

export class ApiService {
  #baseURL;
  #timeout;

  constructor(baseURL, timeout = 8000) {
    this.#baseURL  = baseURL.replace(/\/$/, '');
    this.#timeout  = timeout;
  }

  async #request(endpoint, options = {}) {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), this.#timeout);

    try {
      const res = await fetch(`${this.#baseURL}${endpoint}`, {
        headers: { 'Content-Type': 'application/json' },
        signal: controller.signal,
        ...options,
      });
      clearTimeout(timer);
      if (!res.ok) throw new Error(`HTTP ${res.status}: ${res.statusText}`);
      return res.json();
    } catch (err) {
      clearTimeout(timer);
      if (err.name === 'AbortError') throw new Error('انتهت المهلة');
      throw err;
    }
  }

  get(endpoint)          { return this.#request(endpoint); }
  post(endpoint, body)   { return this.#request(endpoint, { method:'POST',  body: JSON.stringify(body) }); }
  put(endpoint, body)    { return this.#request(endpoint, { method:'PUT',   body: JSON.stringify(body) }); }
  delete(endpoint)       { return this.#request(endpoint, { method:'DELETE' }); }
}

// ═══ js/main.js — Entry Point ═══

import { Student }          from './models/Student.js';
import { StorageService }   from './services/StorageService.js';
import { ApiService }       from './services/ApiService.js';
import { Router }           from './ui/Router.js';
import { Dashboard }        from './ui/Dashboard.js';
import { StudentsPage }     from './ui/StudentsPage.js';

// ══ تهيئة الخدمات ══
const storage = new StorageService('techacademy_');
const api     = new ApiService('https://jsonplaceholder.typicode.com');

// ══ استعادة الطلاب من localStorage ══
async function initializeApp() {
  const savedStudents = storage.load('students', []);
  savedStudents.forEach(data => Student.fromJSON(data));

  // جلب بيانات تجريبية من API إذا لا يوجد طلاب
  if (Student.getTotalCount() === 0) {
    try {
      const users = await api.get('/users');
      users.slice(0, 5).forEach(user => {
        try {
          new Student({
            name:  user.name,
            email: user.email,
            phone: '0501234567', // placeholder
          });
        } catch { /* تخطي البيانات غير الصالحة */ }
      });
      saveStudents();
    } catch (err) {
      console.warn('تعذّر جلب بيانات تجريبية:', err.message);
    }
  }

  renderApp();
}

// ══ حفظ الطلاب ══
function saveStudents() {
  const data = Student.getAll().map(s => s.toJSON());
  storage.save('students', data);
}

// ══ تهيئة الـ Router ══
function renderApp() {
  const router = new Router({
    '#dashboard': () => new Dashboard(Student.getAll()).render(),
    '#students':  () => new StudentsPage(Student.getAll(), { onSave: saveStudents }).render(),
  });

  router.init('#dashboard'); // الصفحة الافتراضية
}

// ══ تشغيل التطبيق ══
document.addEventListener('DOMContentLoaded', initializeApp);

// ══ مزامنة بين التبويبات ══
window.addEventListener('storage', (e) => {
